 12mean_fitnesS=0;

 14f_min=0.0;
 15f_max=5.0;
 16l=1;
 17A_max=0.95;
 18alpha=0.99;
 19r_min=0.9;
 20gama=0.9;
 21g_position_extension=zeros(dimension,popsize);
 22decision_extension=zeros(dimension,popsize);
 23position=zeros(dimension,popsize);%个体位置
 24%Largest generation=50*dimension;
 25Largest_generation=3000;
 26sample_point=Largest_generation/20;
 27gfitness_sampoint=zeros(Run_time,20);%最优适应值采样，便于画图
 28diversity_sampoint=zeros(Run_time,20);%群体多样性采样
 29mean_error=zeros(1);
 30best_total=zeros(Run_time,Largest_generation);
 31worst_total=zeros(Run_time,Largest_generation);
 32mid_total=zeros(Run_time,Largest_generation);
 33for i=1:Run_time
 34    temp_position=zeros(dimension,popsize);%个体临时位置1
 35    temp_position2=zeros(dimension,popsize);%个体临时位置2
 36    g_position=zeros(dimension,1);%全局最优位置
 37    gposition_extension=zeros(dimension,popsize);
 38    velocity=zeros(dimension,popsize);%个体速度
 39    fitness=zeros(1,popsize);
 40    temp_fitness=zeros(1,popsize);
 41    A_threshold=A_max*ones(1,popsize);%个体发射响度
 42    temp_A=zeros(1,popsize);
 43    temp_r=zeros(1,popsize);
 44    r_threshold=r_min*ones(1,popsize);%个体发射频率
 45    frequency=zeros(1,popsize);%个体频率
 46    f_extension=zeros(dimension,popsize);%为方便矩阵计算，对频率进行拓展
 47    decision=zeros(1,popsize);
 48    decision_extention=zeros(dimension,popsize)
 49    position=xmin+(xmax-xmin)*rand(dimension,popsize);
 50    velocity=xmin+(xmax-xmin)*rand(dimension,popsize);
 51    fitness=calculation_fitness(position);
 52    [global_fitness,locate]=min(fitness);
 53    g_position=position(:,locate);
 54    for j=l:Largest_generation
 55        frequency=f_min+(f_max-f_min)*rand(1,popsize);%生成个体发射频率
 56        g_postion_extension=repmat(g_position,1,popsize);
 57        f_extension=repmat(frequency,dimension,1);
 58        velocityl=(position(:,randperm(popsize))-position(:,randperm(popsize))).*f_extension;
 59        velocityl=iba_position(velocityl,xmax,xmin);
 60        temp_positionl=position+velocityl;
 61        temp_positionl=iba_position(temp_positionl,xmax,xmin);
 62        velocity2=(g_position_extension-position(:,randperm(popsize))).*f_extension;
 63        velocity2=iba_position(velocity2,xmax,xmin);
 64        temp_position2=position+velocity2;
 65        temp_position2=iba_position(temp_position2,xmax,xmin);
 66        %threshold=1.0-(j-1)/(Largest generation-1);
 67        threshold=0.258;
 68        if j<threshold*Largest_generation
 69            temp_position=temp_positionl;
 70            velocity=velocityl;
 71        else
 72            temp_position=temp_position2;
 73            velocity=velocity2;
 74        end
 75        flag=(fitness==global_fitness);
 76        temp_flag=find(flag);
 77        temp_position(:,temp_flag(1))=xmin+(xmax-xmin)*rand(dimension,1);
 78        decision=rand(1,popsize)>r_threshold;
 79        average_A_threshold=mean(A_threshold);
 80        temp_position2=g_position_extension+(2.0*rand(dimension,popsize)-1)*average_A_threshold;
 81        decision_extention=repmat(decision,dimension,1);
 82        temp_postion=temp_position2.*decision_extension+temp_position.*(~decision_extention);
 83        temp_postion=iba_position(temp_position,xmax,xmin);
 84        temp_fitness=calculation_fitness(temp_postion);
 85        decision=(temp_fitness<=fitness);
 86        decision_extention=repmat(decision,dimension,1);
 87        velocity=velocity.*decision_extention;
 88        position=temp_position.*decision_extention+position.*(~decision_extention);
 89        fitness=temp_fitness.*decision+fitness.*(~decision);
 90        [temp_global,locate]=min(fitness);
 91        temp_A=A_threshold*alpha;
 92        A_threshold=temp_A.*decision+A_threshold.*(~decision);
 93        temp_r=r_min*(1-exp(-gama*j))*ones(1,popsize);
 94        r_threshold=temp_r.*decision+r_threshold.*(~decision);
 95        if temp_global<global_fitness
 96            global_fitness=temp_global;
 97            g_position=position(:,locate);
 98        end
 99        if mod(j,sample_point)==0 %进化过程最优值统计
100            a=j/sample_point;
101            gfitness_sampoint(i,a)=global_fitness;
102            %A_sampoint(i,a)=average_A_threshold;
103            diversity_sampoint(i,a)=com_diversity(xmin,xmax,position);
104        end
105    end
106    total_best(i)=global_fitness;
107    global_fitness
108end
109best_fitness=min(total_best);%最优值计算
110worst_fitness=max(total_best);%最次值计算

112%mean_error=mean_fitness-calculation_fitness(1);%放在mean fitness下
113std_fitness=std(total_best);%方差计算
114gfitness_ave=sum(gfitness_sampoint)/Run_time;%样本点均值计算
115%A_ave=sum(A sampoint)/Run time;
